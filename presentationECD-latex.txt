% Introduction
\section*{Introduction Générale}
\addcontentsline{toc}{section}{Introduction Générale}

La résolution numérique d'équations différentielles ordinaires (EDO) constitue un domaine fondamental de l'analyse numérique avec des applications vastes en physique, ingénierie, biologie et économie. Ce rapport présente une étude comparative de trois méthodes classiques de résolution d'EDO : Euler explicite, Heun (Euler amélioré) et Runge-Kutta d'ordre 4.

\subsection*{Objectifs du Travail}
\begin{itemize}
    \item Implémenter et tester trois méthodes numériques de résolution d'EDO
    \item Comparer leur précision sur trois équations tests représentatives
    \item Analyser leur efficacité en termes de temps de calcul
    \item Établir des recommandations pratiques pour le choix des méthodes
\end{itemize}

\subsection*{Méthodologie}
\begin{enumerate}
    \item Développement d'une classe Python \texttt{SolveurEDO} implémentant les trois méthodes
    \item Sélection de trois équations différentielles avec solutions analytiques connues
    \item Mesure systématique des temps d'exécution et des erreurs
    \item Visualisation comparative des résultats
\end{enumerate}

\subsection{Analyse Détailée par Équation}

\subsubsection{Équation 1 : Croissance exponentielle modifiée}
\[
z'(x) = 0.1 \cdot x \cdot z(x), \quad z(0) = 1
\]

\begin{resultbox}
    \textbf{Résultats pour l'Équation 1 :}
    \begin{itemize}
        \item \important{Runge-Kutta 4} : Performance exceptionnelle (erreur $\num{4.18e-6}$)
        \item \important{Heun} : Bonne précision (erreur $\num{6.24e-3}$)
        \item \important{Euler} : Précision modérée (erreur $\num{3.55e-1}$)
        \item \textbf{Amélioration RK4/Euler} : \textbf{85 000 fois plus précis}
        \item \textbf{Amélioration Heun/Euler} : \textbf{57 fois plus précis}
    \end{itemize}
\end{resultbox}

\subsubsection{Équation 2 : Problème avec singularité}
\[
z'(x) = \frac{1 - 30x}{2\sqrt{x}} + 15z(x), \quad z(0.01) = \sqrt{0.01}
\]

\begin{resultbox}
    \textbf{Analyse critique - Équation 2 :}
    \begin{itemize}
        \item \warning{Toutes les méthodes présentent des erreurs très importantes}
        \item \important{Euler} : Erreur "relativement" plus faible ($\num{5.30e3}$)
        \item \important{Runge-Kutta 4} : Échec de convergence (erreur $\num{1.28e9}$)
        \item \textbf{Causes probables :}
        \begin{enumerate}
            \item Singularité en $x=0$ mal gérée
            \item Condition initiale problématique ($z(0.01) = 0.1$)
            \item Instabilité numérique due au coefficient $+15z(x)$
        \end{enumerate}
    \end{itemize}
\end{resultbox}

\subsubsection{Équation 3 : Coefficient périodique}
\[
z'(x) = \pi \cos(\pi x) z(x), \quad z(0) = 1
\]

\begin{resultbox}
    \textbf{Résultats pour l'Équation 3 :}
    \begin{itemize}
        \item \important{Runge-Kutta 4} : Meilleure précision (erreur $\num{8.13e-3}$)
        \item \important{Heun} : Performance intermédiaire (erreur $\num{4.77e-1}$)
        \item \important{Euler} : Précision limitée (erreur $\num{7.77e-1}$)
        \item \textbf{Amélioration RK4/Euler} : \textbf{96 fois plus précis}
        \item \textbf{Comportement oscillant} bien capturé par RK4
    \end{itemize}
\end{resultbox}

\section{Interprétation et Discussion}

\subsection{Succès et Échecs}

\subsubsection{Succès Remarquables}

\begin{resultbox}
    \textbf{Cas de réussite : Équation 1}
    \begin{itemize}
        \item \important{RK4} : Erreur $\num{4.18e-6}$ → performance exceptionnelle
        \item \important{Heun} : Erreur $\num{6.24e-3}$ → très bon compromis
        \item \important{Euler} : Acceptable pour applications non critiques
        \item \textbf{Conclusion :} Pour les problèmes réguliers, l'ordre de la méthode est déterminant
    \end{itemize}
\end{resultbox}

\subsubsection{Problèmes Identifiés}

\begin{resultbox}
    \textbf{Cas problématique : Équation 2}
    \begin{itemize}
        \item \warning{Instabilité numérique sévère} pour toutes les méthodes
        \item \important{Euler} : "Meilleure" performance relative
        \item \important{RK4} : Amplification des erreurs (phénomène de stiff problem)
        \item \textbf{Causes :}
        \begin{enumerate}
            \item Terme $+15z(x)$ crée une instabilité exponentielle
            \item Singularité $\frac{1}{2\sqrt{x}}$ mal gérée
            \item Condition initiale $z(0.01)=0.1$ potentiellement incorrecte
        \end{enumerate}
    \end{itemize}
\end{resultbox}

\subsection{Impact du Pas d'Intégration}

\begin{resultbox}
    \textbf{Influence du pas $h=0.3$ :}
    \begin{itemize}
        \item \important{Pas relativement grand} pour une précision fine
        \item \important{RK4} : Tolère mieux les pas grands ($O(h^4)$ vs $O(h)$)
        \item \important{Euler} : Nécessite des pas très petits pour être précis
        \item \textbf{Recommandation :} Pour $h=0.3$, RK4 est clairement supérieur
        \item \textbf{Perspective :} Avec $h=0.01$, les écarts seraient encore plus marqués
    \end{itemize}
\end{resultbox}

\section{Conclusions et Recommandations}

\subsection{Conclusions Principales}

\begin{enumerate}
    \item \important{RK4 confirme sa supériorité} pour les problèmes réguliers (erreur $\num{4.18e-6}$ sur Éq.1)
    \item \important{Heun offre le meilleur compromis} global entre précision et complexité
    \item \important{Euler reste utile} pour sa robustesse et sa simplicité
    \item \important{L'Équation 2 révèle des limites} des méthodes explicites pour les problèmes raides
\end{enumerate}

\subsection{Recommandations Pratiques}

\subsection{Améliorations Proposées}

\subsubsection{Sur le Plan Numérique}

\begin{itemize}
    \item \important{Réduire le pas d'intégration} pour améliorer la précision
    \item \important{Implémenter des méthodes adaptatives} à pas variable
    \item \important{Considérer des méthodes implicites} pour les problèmes raides
    \item \important{Ajouter un contrôle d'erreur} pour détecter les instabilités
\end{itemize}

\subsubsection{Sur le Plan Algorithmique}

\begin{itemize}
    \item \important{Validation des conditions initiales} pour éviter les erreurs
    \item \important{Traitement des singularités} par changement de variable
    \item \important{Optimisation du code} pour les calculs intensifs
    \item \important{Parallélisation} des évaluations de fonctions
\end{itemize}

\subsection{Perspectives de Recherche}

\begin{enumerate}
    \item \textbf{Étude des méthodes adaptatives} : pas variable selon l'erreur locale
    \item \textbf{Analyse de stabilité} : détermination des régions de stabilité
    \item \textbf{Comparaison avec méthodes implicites} : pour problèmes raides
    \item \textbf{Application à des problèmes réels} : physique, ingénierie, biologie
    \item \textbf{Optimisation des performances} : utilisation de bibliothèques numériques
\end{enumerate}

        \item Euler est la méthode la plus rapide mais la moins précise
        \item Heun est un bon compromis entre précision et vitesse
        \item Pour des pas plus petits, toutes les méthodes seraient plus précises
    \end{enumerate}
    
    \textbf{Notre analyse confirme ces conclusions et les nuance :}
    \begin{itemize}
        \item La supériorité de RK4 est confirmée sur les problèmes réguliers
        \item Euler est effectivement la plus rapide mais aussi la plus robuste
        \item Heun représente effectivement le meilleur compromis global
        \item L'importance du pas est critique, surtout pour Euler
    \end{itemize}
\end{resultbox}
