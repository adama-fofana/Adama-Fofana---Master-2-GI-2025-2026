% Introduction
\section*{Introduction Générale}
\addcontentsline{toc}{section}{Introduction Générale}

L'intégration numérique, également appelée quadrature numérique, constitue un pilier fondamental du calcul scientifique. Ce rapport présente une étude comparative exhaustive de cinq méthodes d'intégration numérique appliquées à quatre fonctions tests représentant différents cas d'utilisation.

\subsection*{Objectifs du Travail}
\begin{itemize}
    \item Implémenter et comparer cinq méthodes d'intégration numérique
    \item Tester ces méthodes sur des fonctions avec différents types de poids
    \item Analyser la précision et l'efficacité computationnelle
    \item Établir des recommandations pratiques pour le choix des méthodes
\end{itemize}

\subsection*{Méthodes Étudiées}
\begin{enumerate}
    \item Quadrature de Gauss-Legendre (méthode générale)
    \item Quadrature de Gauss-Laguerre (pour intégrales avec poids $e^{-x}$)
    \item Quadrature de Gauss-Chebyshev (pour poids $1/\sqrt{1-x^2}$)
    \item Méthode composite de Simpson (méthode classique)
    \item Intégration par spline cubique (méthode d'interpolation)
\end{enumerate}

\section{Méthodes d'Intégration Numérique}

\subsection{Quadratures de Gauss}

\subsubsection{Gauss-Legendre}
Pour l'intégrale $\int_a^b f(x) dx$ :
\[
\int_a^b f(x) dx \approx \frac{b-a}{2} \sum_{i=1}^n w_i f\left(\frac{b-a}{2}x_i + \frac{a+b}{2}\right)
\]
où $x_i$ et $w_i$ sont les points et poids de Legendre sur $[-1, 1]$.

\subsubsection{Gauss-Laguerre}
Pour les intégrales sur $[0, \infty)$ avec poids $e^{-x}$ :
\[
\int_0^\infty e^{-x} f(x) dx \approx \sum_{i=1}^n w_i f(x_i)
\]

\subsubsection{Gauss-Chebyshev}
Pour les intégrales sur $[-1, 1]$ avec poids $1/\sqrt{1-x^2}$ :
\[
\int_{-1}^1 \frac{f(x)}{\sqrt{1-x^2}} dx \approx \sum_{i=1}^n w_i f(x_i)
\]

\subsection{Méthode Composite de Simpson}
\[
\int_a^b f(x) dx \approx \frac{h}{3} \left[ f(x_0) + f(x_n) + 4\sum_{i=1}^{n/2} f(x_{2i-1}) + 2\sum_{i=1}^{n/2-1} f(x_{2i}) \right]
\]
avec $h = \frac{b-a}{n}$ et $n$ pair.

\subsection{Intégration par Spline Cubique}
\begin{enumerate}
    \item Échantillonnage de la fonction aux points $x_i$
    \item Construction d'une spline cubique interpolatrice
    \item Intégration analytique de la spline
\end{enumerate}

\section{Discussion Critique des Résultats}

\subsection{Succès et Performances Exceptionnelles}

\begin{itemize}
    \item \important{Gauss-Laguerre} : Convergence spectaculaire sur sa fonction cible
    \item \important{Gauss-Legendre} : Excellence sur fonction standard
    \item \important{Simpson} : Robustesse et simplicité efficaces
\end{itemize}

\subsection{Limitations et Problèmes Identifiés}

\subsubsection{Problème avec la Fonction Chebyshev}

\begin{resultbox}
    \textbf{Anomalie observée :}
    \begin{itemize}
        \item Toutes les méthodes donnent des erreurs importantes ($>0.5$)
        \item La valeur exacte $\num{-7.7262999086e-1}$ semble problématique
        \item \textbf{Hypothèses :}
        \begin{enumerate}
            \item Erreur dans le calcul de la valeur exacte théorique
            \item Problème d'implémentation du poids $1/\sqrt{1-x^2}$
            \item Singularités aux bornes non correctement traitées
        \end{enumerate}
    \end{itemize}
\end{resultbox}

\subsubsection{Limitations Méthodologiques}

\begin{itemize}
    \item \important{Adaptabilité limitée :} Chaque méthode de Gauss est spécialisée
    \item \important{Complexité d'implémentation :} Spline nécessite plus de ressources
    \item \important{Intervalles infinis :} Seule Gauss-Laguerre gère $[0, \infty)$
\end{itemize}

\section{Conclusions et Recommandations}

\subsection{Conclusions Principales}

\begin{enumerate}
    \item \important{Spécialisation payante :} Les méthodes de Gauss adaptées sont imbattables sur leurs domaines
    \item \important{Gauss-Legendre universelle :} Excellente performance sur fonctions standards
    \item \important{Simpson robuste :} Bon compromis simplicité/performance
    \item \important{Spline coûteuse :} Justifiée seulement si interpolation nécessaire
    \item \important{Importance du bon outil :} Le choix de la méthode est critique
\end{enumerate}

\subsection{Recommandations Pratiques}

\begin{table}[H]
    \centering
    \caption{Guide de sélection des méthodes}

        Avec poids $e^{-x}$ sur $[0,\infty)$ & \good{Gauss-Laguerre} & Seule méthode adaptée \\
        Avec poids $1/\sqrt{1-x^2}$ & \good{Gauss-Chebyshev} & Théoriquement optimale \\
        Application simple/générale & \good{Simpson} & Bon compromis global \\
        Besoin d'interpolation & \good{Spline cubique} & Double fonctionnalité \\
        Calcul pédagogique & \good{Simpson} & Simplicité d'explication \\

\subsection{Perspectives d'Amélioration}

\subsubsection{Améliorations Algorithmiques}

\begin{itemize}
    \item \important{Méthodes adaptatives :} Pas variable selon la régularité
    \item \important{Combinaison de méthodes :} Utiliser différentes méthodes par sous-intervalles

\subsection{Code Python Essentiel}

\begin{lstlisting}[caption=Structure de la classe IntegrationNumerique]
class IntegrationNumerique:
    def __init__(self):
        pass
    
    def gauss_legendre(self, f, a, b, n):
        """Quadrature de Gauss-Legendre classique"""
        if b == np.inf:
            def g(t):
                x = t / (1 - t**2) if t != 1 else np.inf
                return f(x) * (1 + t**2) / (1 - t**2)**2
            a_tr = -0.999
            b_tr = 0.999
        else:
            g = f
            a_tr = a
            b_tr = b
        
        x, w = roots_legendre(n)
        t = 0.5 * (b_tr - a_tr) * x + 0.5 * (a_tr + b_tr)
        integral = 0.5 * (b_tr - a_tr) * np.sum(w * g(t))
        return integral
    
    def gauss_laguerre(self, f, n):
        """Quadrature de Gauss-Laguerre pour [0, +∞)"""
        x, w = roots_laguerre(n)
        integral = np.sum(w * f(x))
        return integral
    
    def simpson_composite(self, f, a, b, n):
        """Méthode composite de Simpson"""
        if n % 2 == 1:
            n += 1
        
        h = (b - a) / n
        x = np.linspace(a, b, n+1)
        y = f(x)
        
        integral = h/3 * (y[0] + y[-1] + 
                         4*np.sum(y[1:-1:2]) + 
                         2*np.sum(y[2:-2:2]))
        return integral
\end{lstlisting}
